create function ex79(codvend int, estado char, prazo int)
returns table(codigo int, nome varchar)
language sql
as $$
	select c.codigo_cliente, c.nome_cliente
	from cliente c inner join pedido p
	on c.codigo_cliente=p.codigo_cliente
	where p.codigo_vendedor=codvend and 
	c.uf=estado and 
	p.prazo_entrega=prazo;
$$;
